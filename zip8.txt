import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans, DBSCAN

# Load and preprocess data
df = pd.read_csv("Mall_customers.csv")
df = df.iloc[:, [3, 4]].values

plt.scatter(df[:,0], df[:,1], s=10, c= "black")

# K-Means clustering to determine the optimal number of clusters
wcss = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, init='k-means++', max_iter=300, n_init=10).fit(df)
    wcss.append(kmeans.inertia_)

# Plotting the elbow method graph
plt.plot(range(1, 11), wcss)
plt.title("The Elbow Method")
plt.xlabel("Number of clusters")
plt.ylabel("WCSS")
plt.show()

# Applying DBSCAN
dbscan = DBSCAN(eps=5, min_samples=5).fit(df)
labels = dbscan.labels_

# Visualizing clusters including outliers
colors = ['black', 'blue', 'red', 'green', 'brown', 'pink', 'yellow', 'silver']
for label, color in zip(np.unique(labels), colors):
    plt.scatter(df[labels == label, 0], df[labels == label, 1], s=10, c=color, label=f'Cluster {label}')

plt.xlabel('Annual Income')
plt.ylabel('Spending Score')
plt.legend()
plt.show()
