import sys

max_salary = 0

for line in sys.stdin:
    key, value = line.strip().split('\t', 1)
    try:
        salary= int(value)
    except ValueError:
        continue
    if salary > max_salary:
        max_salary= salary

print("Max Salary: %s" % max_salary)
